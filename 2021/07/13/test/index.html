#include<bits/stdc++.h>
#define mid ((l+r)>>1)
#define P make_pair
using namespace std;
int n,p,q,a,b,c,tot,pois,cnt,trees;
map<int,int>mp;
int bac[40001],m,dep[40001],top[40001],fa[40001];
vector<int>road[40001];
vector<pair<int,int> >has[40001];
int id[40001],siz[40001],w_son[40001];
int val[11560000];
int son[40001*17][2];
int Son[11560000][2];
int rt[40001],rts[40001*17];
int eds[40001];
queue<pair<int,int> >que,tmp;
int read()
{
    int x=0,f=1;
    char c=getchar();
    while(c<'0'||c>'9'){if(c=='-') f=-1;c=getchar();}
    while(c>='0'&&c<='9') x=x*10+c-'0',c=getchar();
    return x*f;
}

void write(int x) {
     if(x<0) putchar('-'),x=-x;
     if(x>9) write(x/10);
     putchar(x%10+'0');
}
struct io{
    int x,xs;
};
queue<io>chek,zj;
bool comp(int a,int b)
{
    return id[a]<id[b];
}
void push_up(int x)
{
    val[x]=val[Son[x][0]]+val[Son[x][1]];
    return ;
}
int fn(int x,int l,int r,int L,int R)
{
    if(l>R||r<L)return 0;
    if(l>=L&&r<=R)return val[x];
    return fn(Son[x][0],l,mid,L,R)+fn(Son[x][1],mid+1,r,L,R);
}
void Add(int &x,int la,int l,int r,int poz)
{
    x=++cnt;
    if(l==r)
    {
        val[x]=val[la]+1;
        return ;
    }
    Son[x][0]=Son[la][0];
    Son[x][1]=Son[la][1];
    if(poz<=mid)
    Add(Son[x][0],Son[la][0],l,mid,poz);
    else
    Add(Son[x][1],Son[la][1],mid+1,r,poz);
    push_up(x);
    return ;
}
void add(int &x,int la,int l,int r,int poz,int val)
{
    x=++pois;
    son[x][0]=son[la][0];
    son[x][1]=son[la][1];
    Add(rts[x],rts[la],1,n,val);
    if(l!=r)
    {
        if(poz<=mid)add(son[x][0],son[la][0],l,mid,poz,val);
        else add(son[x][1],son[la][1],mid+1,r,poz,val);
    }
}
void dfs1(int x,int f)
{
    siz[x]=1;
    dep[x]=dep[f]+1;
    fa[x]=f;
    for(int i=0;i<road[x].size();++i)
    {
        if(road[x][i]!=f)
        {
            dfs1(road[x][i],x);
            siz[x]+=siz[road[x][i]];
            if(siz[road[x][i]]>siz[w_son[x]])
            w_son[x]=road[x][i];
        }
    }
}
void dfs2(int x,int f,int tp)
{
    top[x]=tp;
    id[x]=++tot;
    if(w_son[x])
    dfs2(w_son[x],x,tp);
    for(int i=0;i<road[x].size();++i)
    {
        if(road[x][i]!=f&&road[x][i]!=w_son[x])
        {
            dfs2(road[x][i],x,road[x][i]);
        }
    }
}
void dfs3(int x,int f)
{
    eds[id[x]]=eds[id[f]];
    for(int i=0;i<has[id[x]].size();++i)
    {
        ++trees;
        add(rt[trees],rt[eds[id[x]]],1,m,mp[has[id[x]][i].second],has[id[x]][i].first);
        eds[id[x]]=trees;
    }
    if(w_son[x])
        dfs3(w_son[x],x);
    for(int i=0;i<road[x].size();++i)
    {
        if(road[x][i]!=f&&road[x][i]!=w_son[x])
        {
            dfs3(road[x][i],x);
        }
    }
}
int lca(int x,int y)
{
    while(top[x]!=top[y])
    {
        if(dep[top[x]]<dep[top[y]])swap(x,y);
        x=fa[top[x]];
    }
    return dep[x]<dep[y]?x:y;
}
void ch(int x,int y)
{
    while(que.size())
    que.pop();
    while(top[x]!=top[y])
    {
        if(dep[top[x]]<dep[top[y]])
        swap(x,y);
        que.push(P(id[top[x]],id[x]));
        x=fa[top[x]];
    }
    if(dep[x]>dep[y])swap(x,y);
    que.push(P(id[x],id[y]));
    return ;
}
int query(int t1,int t2,int t3,int t4,int k)
{
    int l=1,r=m;
    while(chek.size())chek.pop();
    chek.push((io){t1,-1});
    chek.push((io){t2,1});
    chek.push((io){t3,-1});
    chek.push((io){t4,1});
    while(l!=r)
    {
        int ans=0;
        while(chek.size())
        {
            int a=chek.front().x,x=chek.front().xs;
            zj.push(chek.front());
            chek.pop();
            while(!que.empty())
            {
                int ll=que.front().first,rr=que.front().second;
                tmp.push(que.front());
                que.pop();
                ans+=x*fn(rts[son[a][0]],1,n,ll,rr);
            }
            swap(tmp,que);
        }
        while(zj.size())
        {
            if(k>ans)
            zj.front().x=son[zj.front().x][1];
            else
            zj.front().x=son[zj.front().x][0];
            chek.push(zj.front());
            zj.pop();
        }
        if(k>ans)
        {
            k-=ans;
            l=mid+1;
        }
        else
        r=mid;
    }
    return bac[l];
}

}
